# 1. 기존 인덱스 삭제 (형식을 새로 맞추기 위해)
DELETE /gold-highlight-scores

# 2. window_start 필드를 'date' 타입으로 미리 지정하기
# 인덱스 템플릿 생성 (설계도 등록)
PUT _index_template/gold_scores_template
{
  "index_patterns": ["gold-highlight-scores*"], # 이 이름으로 시작하면 다 적용!
  "template": {
    "mappings": {
      "properties": {
        "window_start": {
          "type": "date",
          "format": "yyyy-MM-dd HH:mm:ss" # 우리 람다가 쏘는 형식
        },
        "match_id": { "type": "keyword" },
        "final_score": { "type": "double" },
        "event_score": { "type": "double" },
        "chat_score": { "type": "integer" }
      }
    }
  }
}
